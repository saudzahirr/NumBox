cmake_minimum_required(VERSION 3.10)
project(NumBox)

set(CMAKE_CXX_STANDARD 14)

add_subdirectory(LinearAlgebra)
add_subdirectory(RootFindingAlgorithms)
add_subdirectory(Utils)

file(GLOB MAIN_SOURCES "*.cpp")
add_executable(numbox ${MAIN_SOURCES})

target_link_directories(numbox PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/LinearAlgebra ${CMAKE_CURRENT_SOURCE_DIR}/RootFindingAlgorithms ${CMAKE_CURRENT_SOURCE_DIR}/Utils)
